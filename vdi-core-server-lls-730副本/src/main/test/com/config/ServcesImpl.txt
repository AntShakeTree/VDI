package com.vdi.lls.services.impl;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.stereotype.Component;

import com.vdi.common.ParseJSON;
import com.vdi.lls.LLSSendMessage;
import com.vdi.lls.domain.resource.111;
import com.vdi.lls.services.111Service;

@Component("000Service")
public class 111ServiceImpl implements 111Service {
	@Autowired
	@Qualifier("llsHandle")
	private LLSSendMessage llsSendMessage;

	@Override
	public List<111> list111(111 000) {
		111 000Message = new 111();
		000Message.setAction(111.LIST_222_ACTION);
	    000 = (000==null) ? new 111():000;
		000Message.setMapper(000);
		String messsage = ParseJSON.toJson(000Message);
		String result =llsSendMessage.sendMessage(messsage);
		return ParseJSON.parseToJsonArray("content",result, 111.get111ListType());
	}

	@Override
	public 111 get111(String 000Id) {
		111 000 = new 111();
		000.setAction(111.GET_222_ACTION);
		000.set111Identity(000Id);
		String message = ParseJSON.toJson(000);
		String jsonStr = llsSendMessage.sendMessage(message);
		return ParseJSON.fromJson(jsonStr, 111.class);
	}

}
